CREATE OR REPLACE TYPE DIM_OBJ_TYP_PROFESSOR AS OBJECT(
ID_PROFESSOR INTEGER, NOME VARCHAR2(15)
);

CREATE OR REPLACE TYPE DIM_OBJ_TYP_DISCIPLINA AS OBJECT (
ID_DISCIPLINA INTEGER, NOME VARCHAR2(60), SIGLA VARCHAR2(6), AREA VARCHAR2(6), CARGA INTEGER, CREDITOS INTEGER 
);

CREATE OR REPLACE TYPE DIM_OBJ_TYP_CURSO AS OBJECT (
ID_CURSO INTEGER, NOME VARCHAR2(45)
);

CREATE OR REPLACE TYPE DIM_OBJ_TYP_SALA AS OBJECT (
ID_SALA INTEGER, CODIGO VARCHAR2(8), CAPACIDADE INTEGER
);

CREATE OR REPLACE TYPE FAT_OBJ_TYP_ALOCACAO AS OBJECT (
ID_DISCIPLINA REF DIM_OBJ_TYP_DISCIPLINA, ID_PROFESSOR REF DIM_OBJ_TYP_PROFESSOR, ID_CURSO REF DIM_OBJ_TYP_CURSO, COD_TURMA INTEGER, ID_SALA REF DIM_OBJ_TYP_SALA, PERIODO VARCHAR2(6), QTDE_ALUNOS INTEGER, MEDIA_TURMA NUMBER(4,2)
);
