CREATE TABLE DIM_OBJ_TAB_PROFESSOR OF DIM_OBJ_TYP_PROFESSOR (

);

CREATE TABLE DIM_OBJ_TAB_DISCIPLINA OF DIM_OBJ_TYP_DISCIPLINA (
CONSTRAINT DIS_PK PRIMARY KEY (ID_DISCIPLINA)
);

CREATE TABLE DIM_OBJ_TAB_CURSO OF DIM_OBJ_TYP_CURSO (
CONSTRAINT CUR_PK PRIMARY KEY (ID_CURSO)
);

CREATE TABLE DIM_OBJ_TAB_SALA OF DIM_OBJ_TYP_SALA (
CONSTRAINT SALA_PK PRIMARY KEY (ID_SALA)
);
